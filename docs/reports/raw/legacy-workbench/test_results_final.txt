usage: All Options
 -a,--astOut           Print AST
 -b,--asm              Print DLX instructions
 -cfg,--cfg <arg>      Print CFG.dot - requires graphs/
 -i,--in <arg>         Data File
 -loop,--convergence   Run all optimization specified by -o until
                       convergence
 -max,--maxOpt         Run all optimizations till convergence
 -nr,--reg <arg>       Num Regs
 -o,--opt <arg>        Order-sensitive optimization -allowed to have
                       multiple
 -s,--src <arg>        Source File
